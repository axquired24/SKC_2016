-- Keluarkan Semua Peserta
SELECT peserta.nama AS "Nama Peserta", kontingen_all.isi_kontingen AS "Kontingen", peserta.tgl_lahir AS "Tgl Lahir", peserta.berat_badan AS "Berat (Kg)", peserta.perguruan AS "Perguruan", kelas_all.isi_kelas AS "Kelas", peserta.jk AS "Jenis Kelamin"
											FROM peserta INNER JOIN kelas_all
											ON kelas_all.id_kelas=peserta.id_kelas
											INNER JOIN kontingen_all
											ON kontingen_all.id_kontingen=peserta.id_kontingen

-- Keluarkan Jumlah Perkelas
SELECT kelas_all.isi_kelas AS Kelas, COUNT(peserta.nama) AS Jumlah 
									FROM kelas_all INNER JOIN peserta 
									ON kelas_all.id_kelas=peserta.id_kelas 									 
									GROUP BY isi_kelas
